function switchTheme(themeName) {
    const demoBox = document.getElementById('theme-demo-box');

    switch(themeName) {
        case 'light':
            demoBox.style.setProperty('--theme-bg', '#ffffff');
            demoBox.style.setProperty('--theme-text', '#333333');
            demoBox.style.setProperty('--theme-border', '#dddddd');
            break;
        case 'dark':
            demoBox.style.setProperty('--theme-bg', '#1a1a1a');
            demoBox.style.setProperty('--theme-text', '#f0f0f0');
            demoBox.style.setProperty('--theme-border', '#444444');
            break;
        case 'custom':
            demoBox.style.setProperty('--theme-bg', 'linear-gradient(135deg, #667eea, #764ba2)');
            demoBox.style.setProperty('--theme-text', '#ffffff');
            demoBox.style.setProperty('--theme-border', 'transparent');
            break;
    }
}
